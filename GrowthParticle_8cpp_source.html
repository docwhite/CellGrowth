54  // Checks length of children particle list to see if the max particle threshold
55  // is reached or not.
56  if (m_connectedParticles.size() >= m_childrenTreshold) return false;
57 
58  // Creating a list of particles for new particles, those will represent the
59  // branches between the particles
60  std::vector<uint> newConnectedParticles;
61  // Appends mother ID, which is always the first element in the
62  // connectedParticle vector
63  newConnectedParticles.push_back(m_ID);
64 
65  float input0A;
66  float input0B;
67  float input1A;
68  float input1B;
69  float input2A;
70  float input2B;
71 
72  // If it isn't meant to grow towards the light it will chose a random value
73  // between the parents position and twice that value.
74  if (_growToLight==false)
75  {
76  input0A = m_pos[0] + 0.001;
77  input0B = m_pos[0] + m_pos[0];
78  input1A = m_pos[1] + 0.001;
79  input1B = m_pos[1] + m_pos[1];
80  input2A = m_pos[2] + 0.001;
81  input2B = m_pos[2] + m_pos[2];
82  }
83  // If the particle is meant to grow towards the light it will find a value
84  // between the mothers position and the light.
85  else
86  {
87  input0A = m_pos[0] + 0.001;
88  input0B = _lightPos[0];
89  input1A = m_pos[1] + 0.001;
90  input1B = _lightPos[1];
91  input2A = m_pos[2] + 0.001;
92  input2B = _lightPos[2];
93  }
94 
95  // Finding a random position in the predefined boundaries.
96  std::uniform_real_distribution<float> distributionX(input0A, input0B);
97  std::uniform_real_distribution<float> distributionY(input1A, input1B);
98  std::uniform_real_distribution<float> distributionZ(input2A, input2B);
99 
100  QVector3D pos;
101 
102  uint counter = 0;
103  float branchMultiplier = 1.05;
104 
105  do {
106  pos[0] = distributionX(_gen);
107  pos[1] = distributionY(_gen);
108  pos[2] = distributionZ(_gen);
109 
110  QVector3D direction;
111  direction[0]=pos[0]-m_pos[0];
112  direction[1]=pos[1]-m_pos[1];
113  direction[2]=pos[2]-m_pos[2];
114 
115  // Place new particle in direction of vector mutilplied by size of particle.
116  direction.normalize();
117 
118  pos[0] = m_pos[0] + direction[0] * (m_size + m_branchLength + branchMultiplier);
119  pos[1] = m_pos[1] + direction[1] * (m_size + m_branchLength + branchMultiplier);
120  pos[2] = m_pos[2] + direction[2] * (m_size + m_branchLength + branchMultiplier);
121 
122  // Increases the length of a branch the particle is still colliding after 50 tries.
123  if(counter % 50 == 0)
124  {
125  branchMultiplier += 0.5;
126  }
127  counter++;
128  }
129  while(collision(4, pos, _particleList));
130

132  _particleList.push_back(

135  )
136  );
