64 void SpotLight::processMouseMovement(float _offsetx,
65  float _offsety,
66  float _offsetz,
67  QMatrix4x4 _view,
68  QMatrix4x4 _proj)
69 {
70 
71  GLfloat scaleModifier = 0.03;
72  GLfloat w = (_proj * _view * m_model * QVector4D(0,0,0,1)).w();
73  w *= scaleModifier;
74 
75  QVector3D camRight=QVector3D(_view(0,0), _view(0,1), _view(0,2));
76  QVector3D camUp=QVector3D(_view(1,0), _view(1,1), _view(1,2));
77  QVector3D camFront=QVector3D(_view(2,0), _view(2,1), _view(2,2));
78  QVector3D movement;
79  switch(m_manip.getClickedAxis())
80  {
81  case DIRECTION_X:
82  {
83  // Caluculate how much of each offset should be used to move the light.
84  // Dot product will be negative if vectors are moving towards each other,
85  // so flipping the vectors is not necessary.
86  float dotprodx = QVector3D::dotProduct(m_x, camRight);
87  float dotprody = QVector3D::dotProduct(m_x, camUp);
88  float dotprodz = QVector3D::dotProduct(m_x, camFront);
89  float offset=(dotprodx*_offsetx + dotprody*_offsety + dotprodz*_offsetz)*w;
90 
91  movement = offset*m_sensitivity*m_x;
92  break;
93  }
94 
95 
96  case DIRECTION_Y:
97  {
98  float dotprodx = QVector3D::dotProduct(m_y, camRight);
99  float dotprody = QVector3D::dotProduct(m_y, camUp);
100  float dotprodz = QVector3D::dotProduct(m_y, camFront);
101  float offset=(dotprodx*_offsetx + dotprody*_offsety + dotprodz*_offsetz)*w;
102 
103  movement = offset*m_sensitivity*m_y;
104  break;
105  }
106 
107 
108  case DIRECTION_Z:
109  {
110 
111  float dotprodx = QVector3D::dotProduct(m_z, camRight);
112  float dotprody = QVector3D::dotProduct(m_z, camUp);
113  float dotprodz = QVector3D::dotProduct(m_z, camFront);
114  float offset=(dotprodx*_offsetx + dotprody*_offsety + dotprodz*_offsetz)*w;
115 
116  movement = offset*m_sensitivity*m_z;
117  break;
118  }
119  }
120  m_position+=movement;
121  updateModelMatrix();
122 
123  rotate(_offsetx, _offsety);
124 }

126 QQuaternion SpotLight::createFromAngle(const double &_ax, const double &_ay,
127  const double &_az, const double &_offset)
128 {
129  // Here we calculate the sin( theta / 2) once for optimization
130  double factor = sin( _offset / 2.0 );
131 
132  // Calculate the x, y and z of the quaternion
133  double x = _ax * factor;
134  double y = _ay * factor;
135  double z = _az * factor;
136 
137  // Calcualte the w value by cos( theta / 2 )
138  double w = cos( _offset / 2.0 );
139 
140  QQuaternion ret = QQuaternion(w, x, y, z);
141  ret.normalize();
142  return ret;
143 }
