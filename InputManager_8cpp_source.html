172 void InputManager::mouseMoveEvent(QMouseEvent *_event)
173 {
174  GLfloat xpos = _event->pos().x();
175  GLfloat ypos = _event->pos().y();
176  GLfloat xoffset = xpos - m_lastX;
177  GLfloat yoffset = m_lastY - ypos;
178  getUniqueColour(xpos, ypos);
179 
180  // Only process movement if the mouse button and alt is pressed
181  if (m_mousePressed && m_alt_key==true)
182  {
183  m_camera.processMouseMovement(xoffset, yoffset);
184  m_view.setToIdentity();
185  m_view = m_camera.getViewMatrix();
186  }
187  else if(m_mousePressed)
188  {
189  for(auto &s : m_objectList)
190  {
191  // Get manipulator positions
192  QVector3D mp = s->getPosition();
193  m_clickZ=mp.z();
194  float zoffset=xoffset;
195  // Process mouse movement in light class
196  s->processMouseMovement(
197  xoffset,
198  yoffset,
199  zoffset,
200  m_view,
201  m_projection);
202  }
203  }
204 
205  // HOVER
206  for(auto &s : m_objectList)
207  {
208  if(!m_mousePressed)
209  {
210  QOpenGLShaderProgram* m_manipulatorProgram;
211  s->getMainProgram(&m_manipulatorProgram);
212  onHover();
213  }
214  }
215 
216  m_lastX = xpos;
217  m_lastY = ypos;
218 }
