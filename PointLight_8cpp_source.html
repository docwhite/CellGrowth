72 void PointLight::processMouseMovement(float _offsetx,
73  float _offsety,
74  float _offsetz,
75  QMatrix4x4 _view,
76  QMatrix4x4 _proj)
77 {
78 
79  GLfloat scaleModifier = 0.03;
80  GLfloat w=(_proj * _view * m_model * QVector4D(0,0,0,1)).w();
81  w *= scaleModifier;
82 
83  QVector3D camRight=QVector3D(_view(0,0), _view(0,1), _view(0,2));
84  QVector3D camUp=QVector3D(_view(1,0), _view(1,1), _view(1,2));
85  QVector3D camFront=QVector3D(_view(2,0), _view(2,1), _view(2,2));
86  QVector3D movement;
87  switch(m_manip.getClickedAxis())
88  {
89  case DIRECTION_X:
90  {
91  // Caluculate how much of each offset should be used to move the light.
92  // Dot product will be negative if vectors are moving towards each other,
93  // so flipping the vectors is not necessary.
94  QVector3D right = QVector3D(1,0,0);
95  float dotprodx = QVector3D::dotProduct(right, camRight);
96  float dotprody = QVector3D::dotProduct(right, camUp);
97  float dotprodz = QVector3D::dotProduct(right, camFront);
98  float offset=(dotprodx*_offsetx + dotprody*_offsety + dotprodz*_offsetz)*w;
99 
100  movement = offset*m_sensitivity*right;
101  break;
102  }
103 
104 
105  case DIRECTION_Y:
106  {
107  QVector3D up = QVector3D(0,1,0);
108  float dotprodx = QVector3D::dotProduct(up, camRight);
109  float dotprody = QVector3D::dotProduct(up, camUp);
110  float dotprodz = QVector3D::dotProduct(up, camFront);
111  float offset=(dotprodx*_offsetx + dotprody*_offsety + dotprodz*_offsetz)*w;
112 
113  movement = offset*m_sensitivity*up;
114  break;
115  }
116 
117  case DIRECTION_Z:
118  {
119 
120  QVector3D front = QVector3D(0,0,1);
121  float dotprodx = QVector3D::dotProduct(front, camRight);
122  float dotprody = QVector3D::dotProduct(front, camUp);
123  float dotprodz = QVector3D::dotProduct(front, camFront);
124  float offset=(dotprodx*_offsetx + dotprody*_offsety + dotprodz*_offsetz)*w;
125 
126  movement = offset*m_sensitivity*front;
127  break;
128  }
129  }
130  m_position+=movement;
131  updateModelMatrix();
132 }
