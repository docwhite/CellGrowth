42 void LinkedParticle::calculate(
43  std::vector<std::unique_ptr<Particle>> &_particleList,
44  QVector3D _averageDistance,
45  int _cohesionFactor,
46  int _localCohesionFactor,
47  bool _particleDeath)
48 {
49  unsigned int connectionCount = getConnectionCount();
50  std::vector<QVector3D> linkPosition;
51  QVector3D origin;
52 
53  // EQUIDISTANCE
54  // Calculates average distance from centre.
55  // Encourages particles towards this distance from centre.
56  // Stops the particles from wanting to get too close to the middle.
57  QVector3D distance = origin - m_pos;
58  QVector3D m_averageDistance = _averageDistance;
59 
60  if (distance.lengthSquared() < m_averageDistance.lengthSquared())
61  {
62  QVector3D sendAway = -(distance);
63  sendAway /= 100.0;
64  m_vel += sendAway;
65  }
66  else
67  {
68  m_vel /= 1.5;
69  }
70 
71  // COHESION
72  // Calculates cohesion based on all particles.
73  // Sends particles towards particle centre based on distance from centre.
74  QVector3D cohesion = origin - m_pos;
75  float cohesionLength = cohesion.length();
76  float cohesionDist = m_size + (cohesionLength / 2.0);
77  if (cohesionLength >= m_size * 2.0)
78  {
79  m_vel /= 1.1;
80  }
81 
82  cohesion.normalize();
83  cohesion *= (cohesionDist / (_cohesionFactor * 3.3f));
84  m_vel += cohesion;
85 
86  // LOCAL COHESION
87  // Calculates cohesion based on particles links.
88  // Finds the centre of the linked particles.
89  // Influences all particles towards that centre.
90  QVector3D connectionCentre;
91  getPosFromConnections(linkPosition, _particleList);
92 
93  for (uint i = 0; i < connectionCount; i++)
94  {
95  connectionCentre += linkPosition[i];
96  }
97  connectionCentre = connectionCentre/connectionCount;
98  QVector3D localCohesion = connectionCentre - m_pos;
99 
100  float localCohesionLength = localCohesion.length();
101  float localCohesionDist = m_size+(localCohesionLength/2);
102 
103  if (localCohesionLength>=m_size*2)
104  {
105  m_vel/=1.1;
106  }
107 
108  localCohesion.normalize();
109  localCohesion *= (localCohesionDist / (_localCohesionFactor));
110  m_vel += localCohesion;
111 
112  // CALCULATE UNLINKED
113  // Makes a call to calculate unlinked function
114  calculateUnlinked(_particleList);
115 
116  // PARTICLE LIFE
117  // Determines how long the particle has been alive.
118  // Sets velocity to 0 if particle has been alive too long.
119  if (_particleDeath == true)
120  {
121  m_particleLife++;
122  for (uint i = 0; i < connectionCount; i++)
123  {
124  QVector3D distanceFromLinkedParticles = linkPosition[i] - m_pos;
125  if (m_particleLife >= 200
126  && distanceFromLinkedParticles.length() <= (m_size*2))
127  {
128  m_vel.setX(0.0);
129  m_vel.setY(0.0);
130  m_vel.setZ(0.0);
131  }
132  }
133  }
134 }
