297 void Manipulator::createSingleCircle(
298  uint _segments,
299  float _thickness,
300  float _r,
301  int _axis,
302  int _numb,
303  QVector3D _op,
304  std::vector<QVector3D> &_vertices)
305 {
306 
307  float amt=(2*M_PI)/_segments;
308  QVector3D lastPoint = QVector3D(_op.x(),_op.y(),_op.z());
309 
310  for(uint i=0; i<=_segments;i++)
311  {
312 
313  float x=0;
314  float y=0;
315  float z=0;
316 
317  float lx=1;
318  float ly=1;
319  float lz=1;
320  float angle = amt*i;
321 
322 
323  float off_x=0;
324  float off_y=0;
325  float off_z=0;
326  switch(_axis)
327  {
328 
329  case ROTATION_X:
330  {
331  z=_r*cos(angle);
332  y=_r*sin(angle);
333  lz=_thickness;
334  ly=_thickness;
335 
336  if(_numb==1)
337  {
338  x-=(_thickness-1)/2;
339  off_x=(_thickness-1);
340  lz=1;
341  ly=1;
342  }
343  break;
344  }
345 
346  // Around Y axis
347  case ROTATION_Y:
348  {
349  x=_r*cos(angle);
350  z=_r*sin(angle);
351  lx=_thickness;
352  lz=_thickness;
353 
354  if(_numb==1)
355  {
356  y-=(_thickness-1)/2;
357  off_y=(_thickness-1);
358  lx=1;
359  lz=1;
360  }
361 
362 
363  break;
364  }
365  // Around Z axis
366  case ROTATION_Z:
367  {
368  x=_r*cos(angle);
369  y=_r*sin(angle);
370  lx=_thickness;
371  ly=_thickness;
372 
373  if(_numb==1)
374  {
375  z-=(_thickness-1)/2;
376  off_z=(_thickness-1);
377  lx=1;
378  ly=1;
379  }
380  break;
381  }
382  }
383  if(lastPoint==QVector3D(_op.x(),_op.y(),_op.z()))
384  {
385  lastPoint=QVector3D(x,y,z);
386  continue;
387  }
388 
389  _vertices.push_back(QVector3D(x,y,z) +QVector3D(_op.x(),_op.y(),_op.z()));
390  _vertices.push_back(lastPoint +QVector3D(_op.x(),_op.y(),_op.z()));
391  _vertices.push_back(QVector3D(lx, ly, lz)*lastPoint +QVector3D(_op.x()+off_x,_op.y()+off_y,_op.z()+off_z));
392 
393  _vertices.push_back(QVector3D(x,y,z) +QVector3D(_op.x(),_op.y(),_op.z()));
394  _vertices.push_back(QVector3D(lx, ly, lz)*lastPoint +QVector3D(_op.x()+off_x,_op.y()+off_y,_op.z()+off_z));
395  _vertices.push_back(QVector3D(x,y,z)*QVector3D(lx, ly, lz) +QVector3D(_op.x()+off_x,_op.y()+off_y,_op.z()+off_z));
396 
397  lastPoint=QVector3D(x,y,z);
398 
399 
400  }

506  // Vector of normals. To be sent to the shader.

515

519  float s = radius * sin(angle);
520  float c = radius * cos(angle);
521 
522  float angleNext = (2 * M_PI) * ((float)(i + 1) / (float)sectors);
523  float s2 = radius * sin(angleNext);
524  float c2 = radius * cos(angleNext);
525 
526  float x;
527  float y;
528  float z;
529 
530  float x2;
531  float y2;
532  float z2;
533 
534  switch(_axis){
535  case DIRECTION_X:
536  topPoint = QVector3D(_offsetPos.x()+height,_offsetPos.y(),_offsetPos.z());
537  x = 0 + _offsetPos.x();
538  y = s + _offsetPos.y();
539  z = c + _offsetPos.z();
540 
541  x2 = 0 + _offsetPos.x();
542  y2 = s2 + _offsetPos.y();
543  z2 = c2 + _offsetPos.z();
544  break;
545 
546  case DIRECTION_Y:
547  topPoint = QVector3D(_offsetPos.x(), _offsetPos.y() + height, _offsetPos.z());
548  x = c + _offsetPos.x();
549  y = 0 + _offsetPos.y();
550  z = s + _offsetPos.z();
551 
552  x2 = c2 + _offsetPos.x();
553  y2 = 0 + _offsetPos.y();
554  z2 = s2 + _offsetPos.z();
555  break;
556 
557  case DIRECTION_Z:

559  x = s + _offsetPos.x();
560  y = c + _offsetPos.y();
561  z = 0 + _offsetPos.z();
562 
563  x2 = s2 + _offsetPos.x();
564  y2 = c2 + _offsetPos.y();
565  z2 = 0 + _offsetPos.z();
566  break;
567  }
568 
569  QVector3D normal = calculateNormal(QVector3D(x, y, z),topPoint, QVector3D(x2, y2, z2));
570

572  normals.push_back(normal);
573

579  }

583  arrow.numberOfPoints = vertices.size() + normals.size();
584  arrow.uniqueColour=_uniqueColour;
