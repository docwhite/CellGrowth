126  // Set a new front vector since the rotation
127  // will be used on an indentity matrix.
128  QVector3D front=QVector3D(0,0,1);
129  QVector3D pos = getPosition();
130  QVector3D forwardVector = m_rotationPoint-pos;
131  forwardVector.normalize();
132 
133  float dot = QVector3D::dotProduct(front,forwardVector);
134 
135  if ( dot > 0.99999f)
136  {
137  // Rotation point is already framed.
138  return;
139  }
140 
141  // Get the absolute rotation angle.
142  double rotAngle = acos(dot);
143  // Get the axis we need to rotate around.
144  QVector3D rotAxis = QVector3D::crossProduct(front,forwardVector);
145  rotAxis.normalize();
146  // Create quaternion using absolute angle.
147  QQuaternion rotq=createFromAngle(
148  rotAxis.x(),
149  rotAxis.y(),
150  rotAxis.z(),
151  -rotAngle);

153 
154  m_view.setToIdentity();
155  m_view.rotate(rotq);
156 
157
